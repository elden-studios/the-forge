```python
def _mirror_path_for(primary_path):
    """Return the assets/ sibling path for a forge-decisions.json file.

    Mirrors the pattern used by the simulation driver in
    scripts/cabinet_framing_simulate.py.
    """
    parent = os.path.dirname(os.path.abspath(primary_path)) or "."
    basename = os.path.basename(primary_path)
    return os.path.join(parent, "assets", basename)


def close_decision_persist(path, decision_id, new_status="committed"):
    """Load-mutate-atomic-write-mirror wrapper around close_decision.

    Raises ValueError if the decision doesn't exist or is already in a
    terminal state (delegates to close_decision for the terminal-state check).
    """
    with open(path) as f:
        doc = _json.load(f)
    close_decision(doc, decision_id, new_status=new_status)  # pure fn — raises on bad state
    _atomic_write_json(path, doc)
    mirror = _mirror_path_for(path)
    if os.path.isdir(os.path.dirname(mirror)):
        _atomic_write_json(mirror, doc)


def reverse_decision_persist(path, decision_id, successor_id, reason):
    """Load-mutate-atomic-write-mirror wrapper around reverse_decision."""
    with open(path) as f:
        doc = _json.load(f)
    reverse_decision(doc, decision_id, successor_id=successor_id, reason=reason)
    _atomic_write_json(path, doc)
    mirror = _mirror_path_for(path)
    if os.path.isdir(os.path.dirname(mirror)):
        _atomic_write_json(mirror, doc)
```

```python
    return [d for d in doc.get("decisions", []) if d.get("project_id") == project_id]


def query_by_status(doc, status):
    """Return decisions with a given status.

    Raises ValueError if status is not one of DECISION_STATUSES.
    """
    if status not in DECISION_STATUSES:
        raise ValueError(f"status must be one of {sorted(DECISION_STATUSES)}, got: {status!r}")
    return [d for d in doc.get("decisions", []) if d.get("status") == status]


def query_due_soon(doc, now, horizon_days=30):
    """Return non-terminal decisions whose review_at is within `horizon_days` of `now`.

    Terminal statuses (committed, reversed) are excluded regardless of review_at.
    `now` must be a timezone-aware datetime (tests pass tz=UTC).
    """
    horizon = now + timedelta(days=horizon_days)
    out = []
    for d in doc.get("decisions", []):
        if d.get("status") in _TERMINAL_STATUSES:
            continue
        try:
            review_dt = _parse_iso(d.get("review_at"))
        except (ValueError, TypeError, AttributeError):
            continue
        # Normalize to the same tz as `now` for comparison
        if review_dt.tzinfo is None and now.tzinfo is not None:
            review_dt = review_dt.replace(tzinfo=now.tzinfo)
        if review_dt <= horizon:
            out.append(d)
    return out


def query_sorted_by_review_at(doc, reverse=False):
    """Return decisions sorted by review_at (ascending by default).

    Entries with missing/invalid review_at sort to the end regardless of `reverse`.
    """
    def key(d):
        try:
            return (0, _parse_iso(d.get("review_at")))
        except (ValueError, TypeError, AttributeError):
            return (1, datetime.max.replace(tzinfo=timezone.utc))
    return sorted(doc.get("decisions", []), key=key, reverse=reverse)
```

```python
def heatmap_buckets(pre_mortem_items):
    """Aggregate pre-mortem items into a 25-cell 5×5 heatmap (likelihood × impact).

    Returns a dict keyed by (likelihood, impact) tuples, each in the range
    1..5, with values = count of items at that cell. Items with out-of-range
    likelihood or impact are silently dropped. All 25 keys are always present
    in the output, even for empty input — simplifies rendering code.

    The JS dashboard widget mirrors this logic exactly — any divergence is a bug.
    """
    buckets = {(l, i): 0 for l in range(1, 6) for i in range(1, 6)}
    for item in pre_mortem_items:
        l = item.get("likelihood")
        i = item.get("impact")
        if not isinstance(l, int) or not isinstance(i, int):
            continue
        if 1 <= l <= 5 and 1 <= i <= 5:
            buckets[(l, i)] += 1
    return buckets
```

```javascript
function renderCabinetBlock(state, tasks, decisionsDoc) {
  const block = document.getElementById('cabinet-block');
  const empty = document.getElementById('cabinet-empty');
  const framing = tasks?.cabinet_framing;
  const hasFraming = framing && framing.lenses && Object.keys(framing.lenses).length > 0;
  if (!hasFraming) {
    block.style.display = 'none';
    empty.style.display = 'block';
    return;
  }
  block.style.display = 'block';
  empty.style.display = 'none';

  // Verdict — latest open decision for current_project
  const proj = state?.current_project;
  const decisions = (decisionsDoc?.decisions || []).filter(d => d.project_id === proj && d.status === 'open');
  const latest = decisions.sort((a, b) => (a.decided_at < b.decided_at ? 1 : -1))[0];
  const verdictEl = document.getElementById('cabinet-verdict');
  if (latest) {
    const decider = state.agents?.find(a => a.id === latest.decided_by)?.name || latest.decided_by;
    const dissenters = (latest.dissenting || [])
      .map(id => state.agents?.find(a => a.id === id)?.name || id).join(', ');
    verdictEl.innerHTML = `
      <div class="title">${esc(latest.title)}</div>
      <div class="meta">
        <span>Decider: <b>${esc(decider)}</b></span>
        <span>Reversibility: <b>${esc(latest.reversibility)}</b></span>
        <span>Review: ${new Date(latest.review_at).toLocaleDateString()}</span>
        ${dissenters ? `<span class="dissent">Dissent: ${esc(dissenters)}</span>` : ''}
      </div>
    `;
  } else {
    verdictEl.innerHTML = '<div class="muted">No verdict recorded yet.</div>';
  }

  // 5 lenses
  const lensDefs = [
    ['strategic_kernel', 'Strategic Kernel', 'Flint'],
    ['product_shape', 'Product Shape', 'Cade'],
    ['build_class', 'Build Class', 'Helix'],
    ['economic_shape', 'Economic Shape', 'Prism'],
    ['market_bet', 'Market Bet', 'Dune'],
  ];
  const lensesEl = document.getElementById('cabinet-lenses');
  lensesEl.innerHTML = lensDefs.map(([key, title, exec]) => {
    const lens = framing.lenses[key];
    const body = lens?.summary || lens?.body || '—';
    return `<div class="lens-card">
      <div class="lens-title">${esc(title)}</div>
      <div class="lens-exec">${esc(exec)}</div>
      <div class="lens-body">${esc(body)}</div>
    </div>`;
  }).join('');

  // Top 3 pre-mortem risks
  const preMortem = tasks.pre_mortem || [];
  const top3 = [...preMortem].sort((a, b) => (b.likelihood * b.impact) - (a.likelihood * a.impact)).slice(0, 3);
  const risksEl = document.getElementById('cabinet-risks');
  risksEl.innerHTML = top3.map(r => {
    const score = r.likelihood * r.impact;
    const owner = state.agents?.find(a => a.id === r.owner_agent)?.name || r.owner_agent;
    return `<div class="risk-row">
      <span class="risk-score">${score}</span>
      <span>${esc(r.failure_mode)}</span>
      <span style="color:var(--muted)">(${esc(owner)})</span>
    </div>`;
  }).join('') || '<div class="muted">No pre-mortem risks yet.</div>';
}

function esc(s) { return String(s).replace(/[<>&"]/g, c => ({'<':'&lt;','>':'&gt;','&':'&amp;','"':'&quot;'}[c])); }
```

```javascript
function heatmapBuckets(items) {
  // JS mirror of tools/decisions_orchestrator.heatmap_buckets — keep in sync!
  const buckets = {};
  for (let l = 1; l <= 5; l++) for (let i = 1; i <= 5; i++) buckets[`${l},${i}`] = 0;
  for (const it of items || []) {
    const l = it.likelihood, i = it.impact;
    if (Number.isInteger(l) && Number.isInteger(i) && l >= 1 && l <= 5 && i >= 1 && i <= 5) {
      buckets[`${l},${i}`] += 1;
    }
  }
  return buckets;
}

function renderHeatmap(preMortem) {
  const el = document.getElementById('cabinet-heatmap');
  if (!el) return;
  const buckets = heatmapBuckets(preMortem || []);
  let html = '<div class="heatmap-axis-y">Likelihood</div>';
  // Rows: likelihood 5 → 1 (top to bottom), 5 cells each
  for (let l = 5; l >= 1; l--) {
    for (let i = 1; i <= 5; i++) {
      const count = buckets[`${l},${i}`];
      const score = l * i;
      let cls = 'c0';
      if (count > 0) {
        if (score >= 20) cls = 'c25';
        else if (score >= 12) cls = 'c16';
        else if (score >= 8) cls = 'c10';
        else if (score >= 4) cls = 'c5';
        else cls = 'c1';
      }
      html += `<div class="heatmap-cell ${cls}" title="L${l}×I${i} score=${score}">${count || ''}</div>`;
    }
  }
  // X-axis label spans 5 columns (use single cell spanning all)
  html += '<div class="heatmap-axis-x" style="grid-column:2/-1">Impact →</div>';
  el.innerHTML = html;
}
```

```javascript
function exportDecisionsMd(state, decisions) {
  const lines = ['# Decision Log', ''];
  for (const d of decisions) {
    const decider = state.agents?.find(a => a.id === d.decided_by)?.name || d.decided_by;
    lines.push(`## ${d.title}`);
    lines.push('');
    lines.push(`- **Decider:** ${decider}`);
    lines.push(`- **Reversibility:** ${d.reversibility}`);
    lines.push(`- **Status:** ${d.status}`);
    lines.push(`- **Decided at:** ${d.decided_at}`);
    lines.push(`- **Review at:** ${d.review_at || '—'}`);
    lines.push(`- **Project:** ${d.project_id || '—'}`);
    if (d.dissenting?.length) {
      const names = d.dissenting.map(id => state.agents?.find(a => a.id === id)?.name || id).join(', ');
      lines.push(`- **Dissent:** ${names} — ${d.dissent_reason || ''}`);
    }
    lines.push('');
    lines.push(d.context || '');
    lines.push('');
    if (d.alternatives_considered?.length) {
      lines.push('**Alternatives considered:**');
      d.alternatives_considered.forEach(a => lines.push(`- ${a}`));
      lines.push('');
    }
    lines.push('---');
    lines.push('');
  }
  downloadBlob(lines.join('\n'), 'forge-decisions.md', 'text/markdown');
}

function exportDecisionsCsv(state, decisions) {
  const cols = ['id','title','decided_by','decided_by_name','reversibility','status','decided_at','review_at','project_id','dissenting','dissent_reason'];
  const rows = [cols.join(',')];
  for (const d of decisions) {
    const name = state.agents?.find(a => a.id === d.decided_by)?.name || '';
    const dissent = (d.dissenting || []).join(';');
    const row = [
      d.id, d.title, d.decided_by, name, d.reversibility, d.status,
      d.decided_at, d.review_at || '', d.project_id || '',
      dissent, d.dissent_reason || '',
    ].map(csvEscape).join(',');
    rows.push(row);
  }
  downloadBlob(rows.join('\n'), 'forge-decisions.csv', 'text/csv');
}

function exportDecisionsJson(decisionsDoc) {
  downloadBlob(JSON.stringify(decisionsDoc, null, 2), 'forge-decisions.json', 'application/json');
}

function csvEscape(v) {
  const s = String(v ?? '');
  if (s.includes(',') || s.includes('"') || s.includes('\n')) {
    return '"' + s.replace(/"/g, '""') + '"';
  }
  return s;
}

function downloadBlob(content, name, mime) {
  const blob = new Blob([content], { type: mime });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = name; a.click();
  setTimeout(() => URL.revokeObjectURL(url), 1000);
}
```
